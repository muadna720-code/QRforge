  // ===== Build QR Data String =====
  function buildQRData() {
    switch (currentTab) {
      case 'text':
        return textInput.value.trim();
      case 'url':
        return document.getElementById('urlInput').value.trim();
      case 'email': {
        const to = document.getElementById('emailTo').value.trim();
        const subject = document.getElementById('emailSubject').value.trim();
        const body = document.getElementById('emailBody').value.trim();
        if (!to) return '';
        let mailto = `mailto:${to}`;
        const params = [];
        if (subject) params.push(`subject=${encodeURIComponent(subject)}`);
        if (body) params.push(`body=${encodeURIComponent(body)}`);
        if (params.length) mailto += '?' + params.join('&');
        return mailto;
      }
      case 'phone':
        return `tel:${document.getElementById('phoneInput').value.trim()}`;
      case 'wifi': {
        const ssid = document.getElementById('wifiSSID').value.trim();
        const pass = document.getElementById('wifiPass').value;
        const hidden = document.getElementById('wifiHidden').checked;
        if (!ssid) return '';
        let wifi = `WIFI:T:${currentEnc};S:${ssid};`;
        if (currentEnc !== 'nopass') wifi += `P:${pass};`;
        if (hidden) wifi += 'H:true;';
        wifi += ';';
        return wifi;
      }
      case 'vcard': {
        const name = document.getElementById('vcardName').value.trim();
        if (!name) return '';
        const org = document.getElementById('vcardOrg').value.trim();
        const title = document.getElementById('vcardTitle').value.trim();
        const phone = document.getElementById('vcardPhone').value.trim();
        const email = document.getElementById('vcardEmail').value.trim();
        const url = document.getElementById('vcardUrl').value.trim();
        const addr = document.getElementById('vcardAddr').value.trim();
        let vcard = `BEGIN:VCARD\nVERSION:3.0\nFN:${name}`;
        if (org) vcard += `\nORG:${org}`;
        if (title) vcard += `\nTITLE:${title}`;
        if (phone) vcard += `\nTEL:${phone}`;
        if (email) vcard += `\nEMAIL:${email}`;
        if (url) vcard += `\nURL:${url}`;
        if (addr) vcard += `\nADR:;;${addr};;;;`;
        vcard += `\nEND:VCARD`;
        return vcard;
      }
      default:
        return '';
    }
  }

  // ===== Generate QR =====
  function generateQR() {
    const data = buildQRData();
    if (!data) {
      showToast('Please enter some content first', 'error');
      return;
    }

    const ecMap = { L: 1, M: 0, Q: 3, H: 2 };
    const size = parseInt(sizeSlider.value);
    const margin = parseInt(marginSlider.value);
    const fg = fgColor.value;
    const bg = bgColor.value;

    let qr;
    try {
      qr = qrcode(0, ecMap[currentEC]);
      qr.addData(data);
      qr.make();
    } catch (e) {
      showToast('Data too long or contains unsupported characters', 'error');
      return;
    }

    const moduleCount = qr.getModuleCount();
    const totalModules = moduleCount + margin * 2;
    const cellSize = Math.max(1, Math.floor(size / totalModules));
    const actualSize = cellSize * totalModules;

    canvas.width = actualSize;
    canvas.height = actualSize;
    canvas.style.width = Math.min(actualSize, 360) + 'px';
    canvas.style.height = Math.min(actualSize, 360) + 'px';

    // Draw background
    ctx.fillStyle = bg;
    ctx.fillRect(0, 0, actualSize, actualSize);

    // Draw modules with rounded corners
    const radius = Math.max(0, cellSize * 0.3);
    ctx.fillStyle = fg;

    for (let row = 0; row < moduleCount; row++) {
      for (let col = 0; col < moduleCount; col++) {
        if (qr.isDark(row, col)) {
          const x = (col + margin) * cellSize;
          const y = (row + margin) * cellSize;
          const s = cellSize - 0.5; // Slight gap for clarity
          const r = Math.min(radius, s / 2);

          ctx.beginPath();
          ctx.moveTo(x + r, y);
          ctx.lineTo(x + s - r, y);
          ctx.quadraticCurveTo(x + s, y, x + s, y + r);
          ctx.lineTo(x + s, y + s - r);
          ctx.quadraticCurveTo(x + s, y + s, x + s - r, y + s);
          ctx.lineTo(x + r, y + s);
          ctx.quadraticCurveTo(x, y + s, x, y + s - r);
          ctx.lineTo(x, y + r);
          ctx.quadraticCurveTo(x, y, x + r, y);
          ctx.closePath();
          ctx.fill();
        }
      }
    }

    // Show canvas, hide placeholder
    placeholderState.style.display = 'none';
    canvas.style.display = 'block';
    qrPreview.classList.add('has-qr');
    actionButtons.style.display = 'flex';
    infoCard.style.display = 'block';

    // Update info
    const typeLabels = {
      text: 'Plain Text', url: 'URL Link', email: 'Email',
      phone: 'Phone', wifi: 'WiFi Network', vcard: 'vCard Contact'
    };
    document.getElementById('infoType').textContent = typeLabels[currentTab];
    document.getElementById('infoVersion').textContent = Math.ceil((moduleCount - 17) / 4);
    document.getElementById('infoModules').textContent = `${moduleCount} x ${moduleCount}`;
    document.getElementById('infoChars').textContent = data.length;

    currentQRData = data;
    qrGenerated = true;

    // Save to history
    saveToHistory(data, typeLabels[currentTab]);
  }

  // ===== History =====
  function saveToHistory(data, label) {
    // Avoid duplicate of latest
    if (history.length && history[0].data === data) return;

      time: Date.now()

      miniCanvas.width = 36;
      miniCanvas.height = 36;
      const ecMap = { L: 1, M: 0, Q: 3, H: 2 };
      try {
        const miniQR = qrcode(0, ecMap[item.ec] || 0);
        miniQR.addData(item.data);
        miniQR.make();
        const mc = miniQR.getModuleCount();
        const cs = Math.max(1, Math.floor(36 / mc));
        const mctx = miniCanvas.getContext('2d');
        mctx.fillStyle = item.bg || '#0a0a0b';
        mctx.fillRect(0, 0, 36, 36);
        mctx.fillStyle = item.fg || '#e8a849';
        for (let r = 0; r < mc; r++) {
          for (let c = 0; c < mc; c++) {
            if (miniQR.isDark(r, c)) {
              mctx.fillRect(c * cs, r * cs, cs, cs);
            }
          }
        }
      } catch (e) {
        // skip if can't render mini
      }

      const textDiv = document.createElement('div');
      textDiv.className = 'flex-1 min-w-0';

    });

      textDiv.innerHTML = `
        <div class="flex items-center gap-2">
          <span class="text-xs font-semibold" style="color: var(--fg);">${item.label}</span>
          <span class="text-xs" style="color: var(--border);">${timeAgo}</span>
        </div>
        <div class="text-xs truncate" style="color: var(--muted);">${item.data.substring(0, 80)}${item.data.length > 80 ? '...' : ''}</div>
      `;

      div.appendChild(miniCanvas);
      div.appendChild(textDiv);
      div.addEventListener('click', () => {
        historyModal.style.display = 'none';
        loadFromHistory(item);
      });
      historyModalList.appendChild(div);
    });
  }

  function getTimeAgo(timestamp) {
    const diff = Date.now() - timestamp;
    const mins = Math.floor(diff / 60000);
    if (mins < 1) return 'just now';
    if (mins < 60) return `${mins}m ago`;
    const hours = Math.floor(mins / 60);
    if (hours < 24) return `${hours}h ago`;
    const days = Math.floor(hours / 24);
    return `${days}d ago`;
  }

  // ===== Download SVG =====
  document.getElementById('downloadSvg').addEventListener('click', () => {
    if (!qrGenerated) return;
    const size = parseInt(sizeSlider.value);
    const margin = parseInt(marginSlider.value);
    const fg = fgColor.value;
    const bg = bgColor.value;
    const ecMap = { L: 1, M: 0, Q: 3, H: 2 };

    let qr;
    try {
      qr = qrcode(0, ecMap[currentEC]);
      qr.addData(currentQRData);
      qr.make();
    } catch (e) {
      showToast('Failed to generate SVG', 'error');
      return;
    }

    const moduleCount = qr.getModuleCount();
    const totalModules = moduleCount + margin * 2;
    const cellSize = Math.max(1, size / totalModules);

    let svg = `<svg xmlns="http://www.w3.org/2000/svg" width="${size}" height="${size}" viewBox="0 0 ${size} ${size}">`;
    svg += `<rect width="${size}" height="${size}" fill="${bg}"/>`;

    for (let row = 0; row < moduleCount; row++) {
      for (let col = 0; col < moduleCount; col++) {
        if (qr.isDark(row, col)) {
          const x = (col + margin) * cellSize;
          const y = (row + margin) * cellSize;
          const r = Math.min(cellSize * 0.3, cellSize / 2);
          svg += `<rect x="${x}" y="${y}" width="${cellSize}" height="${cellSize}" rx="${r}" ry="${r}" fill="${fg}"/>`;
        }
      }
    }
    svg += '</svg>';

    const blob = new Blob([svg], { type: 'image/svg+xml' });
    const link = document.createElement('a');
    link.download = `qrforge-${Date.now()}.svg`;
    link.href = URL.createObjectURL(blob);
    link.click();
    URL.revokeObjectURL(link.href);
    showToast('SVG downloaded');
  });

    if (!qrGenerated) return;
    try {
      const blob = await new Promise(resolve => canvas.toBlob(resolve, 'image/png'));
      await navigator.clipboard.write([new ClipboardItem({ 'image/png': blob })]);
      showToast('Copied to clipboard');
    } catch (e) {
      // Fallback: copy data string
      try {
        await navigator.clipboard.writeText(currentQRData);
        showToast('QR data copied as text');
      } catch (e2) {
        showToast('Copy not supported in this browser', 'error');
      }
    }
  });

  // ===== Auto-generate on input with debounce =====
  let autoGenTimer;
  function autoGenerate() {
    clearTimeout(autoGenTimer);
    autoGenTimer = setTimeout(() => {
      if (qrGenerated) generateQR();
    }, 500);
  }

  document.querySelectorAll('.input-field').forEach(input => {
    input.addEventListener('input', autoGenerate);
